# -*- coding: utf-8 -*-\
import sys\
import csv\
import os\
from pathlib import Path\
\
from PySide6.QtWidgets import (\
    QApplication, QWidget, QMessageBox,\
    QTableWidgetItem, QHeaderView\
)\
from PySide6.QtUiTools import QUiLoader\
from PySide6.QtCore import QDate, QRegularExpression, QFile, QIODevice\
from PySide6.QtGui import QRegularExpressionValidator\
\
CSV_FILENAME = "personel.csv"\
CSV_DELIMITER = ";"\
\
\
class Widget(QWidget):\
    def __init__(self, parent=None):\
        super().__init__(parent)\
        ui_path = Path("form.ui")\
        if not ui_path.exists():\
            QMessageBox.critical(self, "Aray\'fcz Y\'fckleme Hatas\uc0\u305 ", f"\{ui_path\} bulunamad\u305 .")\
            sys.exit(1)\
\
        loader = QUiLoader()\
        ui_file = QFile(str(ui_path))\
        if not ui_file.open(QIODevice.ReadOnly):\
            QMessageBox.critical(self, "Aray\'fcz Y\'fckleme Hatas\uc0\u305 ", f"\{ui_path\} a\'e7\u305 lamad\u305 .")\
            sys.exit(1)\
        try:\
            self.ui = loader.load(ui_file, parentWidget=self)\
        finally:\
            ui_file.close()\
\
        if not self.ui:\
            QMessageBox.critical(self, "Aray\'fcz Y\'fckleme Hatas\uc0\u305 ", "UI y\'fcklenemedi.")\
            sys.exit(1)\
\
        self.setWindowTitle("Personel Kay\uc0\u305 t D\'fczenleme Ekran\u305 ")\
        self.setup_validators()\
        self.connect_signals()\
        self.load_personnel_table()\
\
#telefon ve tc i\'e7in sadece rakam girilecek \uc0\u351 ekilde ayarlad\u305 m.\
    def setup_validators(self):\
        try:\
            digits_re = QRegularExpression("[0-9]*")\
            digits_validator = QRegularExpressionValidator(digits_re)\
            self.ui.tc_text.setValidator(digits_validator)\
            self.ui.tc_text.setMaxLength(11)\
\
            self.ui.telefon_text.setValidator(digits_validator)\
            self.ui.telefon_text.setMaxLength(11)\
        except AttributeError as e:\
            QMessageBox.critical(self, "Validator Hatas\uc0\u305 ",\
                                 f"UI i\'e7inde beklenen alan bulunamad\uc0\u305 : \{e\}")\
#buton ve checkboxlar\uc0\u305  ayarlad\u305 m.\
    def connect_signals(self):\
        try:\
            # Kaydet\
            self.ui.btn_kaydet.clicked.connect(self.save_data_to_csv)\
            # Tabloyu temizle\
            self.ui.btn_sil.clicked.connect(self.clear_table_data)\
            # CSV'den tekrar y\'fckle\
            self.ui.btn_goruntule.clicked.connect(self.load_personnel_table)\
            # Evet/Hay\uc0\u305 r checkbox'lar\u305 n\u305 n ayarlar\u305 \
            self.ui.e_k.clicked.connect(lambda checked: self.ui.h_k.setChecked(False) if checked else None)\
            self.ui.h_k.clicked.connect(lambda checked: self.ui.e_k.setChecked(False) if checked else None)\
\
        except AttributeError as e:\
            QMessageBox.critical(self, "Ba\uc0\u287 lant\u305  Hatas\u305 ",\
                                 f"Buton/alan isimlerinden biri UI'da yok: \{e\}")\
#verileri csv'ye y\'fckleme.\
    def save_data_to_csv(self):\
        try:\
            user_id = self.ui.id_text.text().strip()\
            ad = self.ui.ad_text.text().strip()\
            soyad = self.ui.soyad_text.text().strip()\
            telefon = self.ui.telefon_text.text().strip()\
            email = self.ui.mail_text.text().strip()\
            tc_kimlik = self.ui.tc_text.text().strip()\
            dogum_tarihi = self.ui.dt_text.date().toString("dd.MM.yyyy")\
\
            if self.ui.e_k.isChecked():\
                aktif_durumu = "Aktif"\
            elif self.ui.h_k.isChecked():\
                aktif_durumu = "Aktif de\uc0\u287 il"\
            else:\
                aktif_durumu = "Belirtilmedi"\
\
            if not user_id or not ad or not soyad:\
                QMessageBox.warning(self, "Eksik Bilgi",\
                                    "User ID, Ad ve Soyad alanlar\uc0\u305  doldurulmal\u305 d\u305 r.")\
                return\
\
            header = ["User ID", "Ad", "Soyad", "Telefon No", "E-Mail",\
                      "TC Kimlik", "Do\uc0\u287 um Tarihi", "Aktif"]\
            new_row = [user_id, ad, soyad, telefon, email, tc_kimlik, dogum_tarihi, aktif_durumu]\
\
            file_exists = os.path.isfile(CSV_FILENAME)\
            write_header = not file_exists or os.path.getsize(CSV_FILENAME) == 0\
\
            with open(CSV_FILENAME, mode="a", newline="", encoding="utf-8-sig") as f:\
                writer = csv.writer(f, delimiter=CSV_DELIMITER)\
                if write_header:\
                    writer.writerow(header)\
                writer.writerow(new_row)\
\
            QMessageBox.information(self, "Ba\uc0\u351 ar\u305 l\u305 ", f"Veri \{CSV_FILENAME\} dosyas\u305 na kaydedildi.")\
            self.load_personnel_table()\
            self.clear_input_fields()\
\
        except Exception as e:\
            QMessageBox.critical(self, "Kay\uc0\u305 t Hatas\u305 ", f"Veri kaydedilirken hata: \{e\}")\
\
#csvdeki verileri tabloya yerle\uc0\u351 tirme\
    def load_personnel_table(self):\
        try:\
            table = self.ui.tbl_personel\
            table.setRowCount(0)\
\
            if not os.path.isfile(CSV_FILENAME):\
                return\
\
            with open(CSV_FILENAME, mode="r", newline="", encoding="utf-8-sig") as f:\
                reader = csv.reader(f, delimiter=CSV_DELIMITER)\
                try:\
                    header = next(reader)\
                except StopIteration:\
                    return\
\
                rows = [row for row in reader if any(cell.strip() for cell in row)]\
\
                if not rows:\
                    table.setColumnCount(len(header))\
                    table.setHorizontalHeaderLabels(header)\
                    table.setRowCount(0)\
                    table.horizontalHeader().setSectionResizeMode(QHeaderView.Stretch)\
                    return\
\
                max_cols = max(len(header), max((len(r) for r in rows), default=0))\
                table.setColumnCount(max_cols)\
                table.setHorizontalHeaderLabels(header + [""] * (max_cols - len(header)))\
                table.setRowCount(len(rows))\
\
                for r_idx, row in enumerate(rows):\
                    for c_idx, value in enumerate(row):\
                        table.setItem(r_idx, c_idx, QTableWidgetItem(value))\
\
            table.horizontalHeader().setSectionResizeMode(QHeaderView.Stretch)\
\
        except Exception as e:\
            QMessageBox.critical(self, "Tablo Y\'fckleme Hatas\uc0\u305 ", f"Tablo y\'fcklenirken hata: \{e\}")\
#tabladaki verileri temizleme\
    def clear_table_data(self):\
        try:\
            self.ui.tbl_personel.setRowCount(0)\
        except Exception as e:\
            QMessageBox.critical(self, "Temizleme Hatas\uc0\u305 ", f"Tablo temizlenirken hata: \{e\}")\
\
    def clear_input_fields(self):\
        try:\
            self.ui.id_text.clear()\
            self.ui.ad_text.clear()\
            self.ui.soyad_text.clear()\
            self.ui.telefon_text.clear()\
            self.ui.mail_text.clear()\
            self.ui.tc_text.clear()\
            self.ui.dt_text.setDate(QDate(2000, 1, 1))\
            self.ui.e_k.setChecked(False)\
            self.ui.h_k.setChecked(False)\
        except Exception as e:\
            print("Alan temizleme hatas\uc0\u305 :", e)\
\
\
if __name__ == "__main__":\
    app = QApplication(sys.argv)\
    main_win = Widget()\
    main_win.show()\
    sys.exit(app.exec())\
